# Running Instructions

    1) git clone 
	2) cd django-weather-app
	3) mkvirtualenv envname
	4) pip install -r requirements.txt
	5) cd main
	6) python manage.py makemigrations
	7) python manage.py migrate
	8) python manage.py runserver
